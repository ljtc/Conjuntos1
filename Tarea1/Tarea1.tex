    \section*{Adicional}
    \setcounter{exercise}{5}
    \begin{exercise}[print=true]
        Sean $X,Y$ conjuntos y $f:X \to Y$. Se define la función $g:\ms{P}(X)\to \ms{P}(Y)$ para cada $a \in \ms{P}(X)$ como $ g(a)=\Set{ y \in Y \st f^{-1}[\{y\}] \subseteq a } $.
        \begin{enumerate}[i)]
            \item Demuestra que si $a \in \ms{P}(X)$ y $b \in \ms{P}(Y)$, entonces $b \subseteq g(a)$ si y sólo si $f^{-1}[b] \subseteq a$.
            \item Prueba que para todo $A \in \ms{P}(\ms{P}(X)) \setminus \{\emptyset\}$ se tiene $g(\bigcap A)=\bigcap \Set{g(a) \st a \in A}$.
        \end{enumerate}
    \end{exercise}
